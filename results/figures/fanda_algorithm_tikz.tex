\begin{tikzpicture}[
  >=Latex,
  font=\small,
  node distance=8mm and 9mm,
  block/.style={
    draw=black!75,
    rounded corners=2pt,
    thick,
    align=center,
    fill=white,
    inner sep=6pt,
    minimum height=10mm,
    text width=34mm
  },
  wideblock/.style={
    block,
    text width=46mm
  },
  io/.style={
    block,
    fill=black!2,
    text width=30mm
  },
  decision/.style={
    draw=black!75,
    diamond,
    aspect=2.1,
    thick,
    align=center,
    fill=white,
    inner sep=2pt,
    text width=22mm
  },
  group/.style={
    draw=black!55,
    rounded corners=4pt,
    dashed,
    inner sep=7pt
  },
  annot/.style={
    align=left,
    font=\footnotesize,
    text width=42mm
  },
  flow/.style={->, thick, draw=black!75},
  sideflow/.style={->, thick, draw=black!55}
]

% Main nodes
\node[io] (prefix0) {Prompt / prefix\\$x_{<t}$};
\node[wideblock, right=of prefix0] (forward0) {Run the model on the current prefix\\and collect layer logits\\$\{z_t^{(\ell)}\}_{\ell \in \mathcal{B}}$, $z_t^{(L)}$};
\node[wideblock, right=of forward0] (select) {Initialization only:\\select anchor layer once\\[2pt]
$\ell^\star = \arg\max_{\ell \in \mathcal{B}}
\mathrm{JSD}\!\left(
\sigma(z_t^{(L)}/\tau),
\sigma(z_t^{(\ell)}/\tau)
\right)$};
\node[block, below=of select] (freeze) {Freeze the anchor\\for the whole answer\\$\ell^\star \leftarrow \mathrm{fixed}$};

\node[wideblock, left=of freeze] (forwardLoop) {For each next token, rerun the model\\on the updated prefix and read\\$z_t^{(L)}$ and $z_t^{(\ell^\star)}$};
\node[wideblock, left=of forwardLoop] (contrast) {Raw-logit contrast\\(no relative-top filter)\\[2pt]
$s_t = z_t^{(L)} - z_t^{(\ell^\star)}$};
\node[block, left=of contrast] (decode) {Choose next token\\$x_t = \arg\max s_t$};
\node[io, below=of decode] (append) {Append token and update prefix\\$x_{<t+1} \leftarrow [x_{<t}; x_t]$};
\node[decision, below=of forwardLoop] (stop) {EOS or\\length limit?};
\node[io, right=of stop] (output) {Final generated answer};

% Arrows
\draw[flow] (prefix0) -- (forward0);
\draw[flow] (forward0) -- (select);
\draw[flow] (select) -- node[right, font=\footnotesize] {step $t=0$} (freeze);
\draw[flow] (freeze) -- (forwardLoop);
\draw[flow] (forwardLoop) -- (contrast);
\draw[flow] (contrast) -- (decode);
\draw[flow] (decode) -- (append);
\draw[flow] (append.east) |- (stop.west);
\draw[flow] (stop.east) -- node[above, font=\footnotesize] {yes} (output.west);
\draw[flow] (stop.north) -- node[right, font=\footnotesize] {no} (forwardLoop.south);

% Loop / init annotations
\node[annot, above=5mm of select] (initnote) {Anchor selection happens once at the beginning. In the released FAnDa variant, this corresponds to an effectively infinite update interval after the first refresh.};
\node[annot, below=5mm of contrast] (contrastnote) {FAnDa keeps DoLa's contrastive idea, but uses the frozen shallow anchor and applies contrast directly in raw-logit space.};

\draw[sideflow] (initnote.south) -- ($(select.north)+(0,1mm)$);
\draw[sideflow] (contrastnote.north) -- ($(contrast.south)+(0,-1mm)$);

% Group boxes
\node[group, fit=(prefix0)(forward0)(select)(freeze), label={[font=\footnotesize, anchor=west]north west:Initialization}] {};
\node[group, fit=(forwardLoop)(contrast)(decode)(append)(stop)(output), label={[font=\footnotesize, anchor=west]north west:Autoregressive decoding loop}] {};

\end{tikzpicture}
